Import inArray from drizzle-orm for batch ID lookup.

Modify the softDelete method to cascade:

```typescript
public async softDelete(id: string, userId: string): Promise<void> {
  const now = new Date();

  // 1. Get all batch IDs for this project (before soft-deleting them)
  const batches = await this.drizzle
    .getClient()
    .select({ id: ingestionBatches.id })
    .from(ingestionBatches)
    .where(
      and(
        eq(ingestionBatches.projectId, id),
        isNull(ingestionBatches.deletedAt),
      ),
    );

  const batchIds = batches.map((b) => b.id);

  // 2. Soft-delete all rows in those batches
  if (batchIds.length > 0) {
    await this.drizzle
      .getClient()
      .update(ingestionRows)
      .set({
        deletedAt: now,
        updatedAt: now,
      })
      .where(
        and(
          inArray(ingestionRows.batchId, batchIds),
          isNull(ingestionRows.deletedAt),
        ),
      );
  }

  // 3. Soft-delete all batches for this project
  await this.drizzle
    .getClient()
    .update(ingestionBatches)
    .set({
      deletedAt: now,
      deletedBy: userId,
      updatedAt: now,
    })
    .where(
      and(
        eq(ingestionBatches.projectId, id),
        isNull(ingestionBatches.deletedAt),
      ),
    );

  // 4. Soft-delete the project itself
  await this.drizzle
    .getClient()
    .update(projects)
    .set({ deletedAt: now, updatedAt: now })
    .where(
      and(
        eq(projects.id, id),
        eq(projects.userId, userId),
        isNull(projects.deletedAt),
      ),
    );
}
```

Note: The operations are not in a transaction here because:
1. This is MVP and CLS transactions would add complexity
2. Soft-delete is idempotent (re-running is safe)
3. If it fails partway, the project record still exists, so a retry will catch remaining items

If strict atomicity is needed later, wrap in a CLS transaction using @Transactional() decorator.
