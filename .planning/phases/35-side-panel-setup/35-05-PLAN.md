**App.tsx -- Add port reconnection with onDisconnect listener:**

Refactor the port lifecycle in the `useEffect` (lines 32-106). Extract port creation into a `connectPort` function that:

1. Creates the port via `chrome.runtime.connect({ name: 'sidepanel' })`
2. Adds the existing message listener to it
3. Adds a `port.onDisconnect` listener that:
   - Logs the disconnection
   - Sets port state to null
   - Waits 500ms (debounce in case of rapid disconnect/reconnect)
   - Calls `connectPort()` again (reconnection)
4. Sends GET_STATE after connection
5. Returns the port

The reconnection should have a maximum retry limit (5 attempts) with exponential backoff (500ms, 1s, 2s, 4s, 8s). After max retries, set error state: "Connection lost. Please reopen the Side Panel."

Use a `useRef` for the port instead of `useState` to avoid stale closure issues in the onDisconnect callback. Keep `useState` for a `portVersion` counter that increments on each reconnection to trigger re-renders when port changes. The ref ensures the reconnect callback always has the latest port reference.

The refactored structure:

```typescript
const portRef = useRef<chrome.runtime.Port | null>(null);
const [portVersion, setPortVersion] = useState(0);
const retriesRef = useRef(0);
const maxRetries = 5;

useEffect(() => {
  let disposed = false;

  function connectPort() {
    if (disposed) return;

    try {
      const p = chrome.runtime.connect({ name: 'sidepanel' });
      portRef.current = p;
      setPortVersion((v) => v + 1);

      p.onMessage.addListener(messageListener);

      p.onDisconnect.addListener(() => {
        console.log('[Sidepanel] Port disconnected, will reconnect...');
        portRef.current = null;

        if (disposed) return;

        retriesRef.current += 1;
        if (retriesRef.current > maxRetries) {
          setError('Connection lost. Please reopen the Side Panel.');
          return;
        }

        const delay = Math.min(500 * Math.pow(2, retriesRef.current - 1), 8000);
        setTimeout(connectPort, delay);
      });

      // Request initial state
      sendViaPort<StateResponse>(p, { type: 'GET_STATE' })
        .then((response) => {
          retriesRef.current = 0; // Reset retries on successful communication
          if (response.success) {
            setState(response.data);
          } else {
            setError(response.error);
          }
          setLoading(false);
        })
        .catch((err) => {
          // If port disconnected during GET_STATE, onDisconnect will handle reconnection
          if (err instanceof PortDisconnectedError) {
            console.log('[Sidepanel] Port disconnected during GET_STATE, awaiting reconnect...');
            return;
          }
          setError(err instanceof Error ? err.message : 'Failed to load state');
          setLoading(false);
        });
    } catch (err) {
      console.error('[Sidepanel] Failed to connect port:', err);
      if (!disposed && retriesRef.current < maxRetries) {
        retriesRef.current += 1;
        const delay = Math.min(500 * Math.pow(2, retriesRef.current - 1), 8000);
        setTimeout(connectPort, delay);
      } else {
        setError('Connection lost. Please reopen the Side Panel.');
      }
    }
  }

  // Define messageListener before connectPort call (existing logic from lines 39-61)
  const messageListener = (message: { type: string; payload?: unknown }) => {
    // ... existing message listener logic unchanged ...
  };

  connectPort();

  // ... existing tab URL query and storage restore logic unchanged ...

  return () => {
    disposed = true;
    portRef.current?.disconnect();
  };
}, []);
```
