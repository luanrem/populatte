2. Add a `handleRecentRowSelect` async function:
```typescript
async function handleRecentRowSelect(rowIndex: number) {
  if (!portRef.current || !state?.batchId) return;

  // Guard: if same row, no-op
  if (rowIndex === state.rowIndex) return;

  // Per locked decision: clicking within same batch = optimistic navigation
  // Update row navigator number instantly
  if (state) {
    setState({ ...state, rowIndex }); // Optimistic update
  }

  // Clear fill results for new row
  setFillResultsMap(new Map());
  setFillError(null);

  try {
    // Send ROW_SELECT to background (created in Plan 01)
    await sendViaPort<VoidResponse>(portRef.current, {
      type: 'ROW_SELECT',
      payload: { rowIndex },
    });
    // State update with real data comes via STATE_UPDATED port message
  } catch (err) {
    console.error('[Sidepanel] Failed to navigate to recent row:', err);
    // Rollback optimistic update by reloading state
    await loadState();
  }
}
```

NOTE: The locked decision mentions "clicking a recent row from a different batch: show confirmation before switching." However, since recent rows are per-batch and the Recentes list only shows entries for the currently selected batch, cross-batch navigation from the Recentes list is not possible. The entries shown are always for `state.batchId`. No confirmation dialog needed for this component.
